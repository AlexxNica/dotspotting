	var url = '' + encodeURIComponent(scan_id) + '/';

	var req = new XMLHttpRequest();
	req.open('GET', url, true);  

	req.onreadystatechange = function (e){  

		if (req.readyState == 4){  
			if (req.status == 200){
				try {
					_show_walkingpaper_scan(req);
				}

				catch (e){ }
			}
		}
	};  

	req.send(null);  
}
